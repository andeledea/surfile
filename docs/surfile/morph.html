 29def _findHfromHist(hist, edges):
 30    """
 31    Finds the 2 maximum values in the histogram and calculates the distance of
 32    the peaks -> gives info about sample step height
 33
 34    Parameters
 35    ----------
 36    hist: np.array
 37        histogram y values
 38    edges: np.array
 39        histogram bins
 40
 41    Returns
 42    ----------
 43    h: float
 44        Height of sample
 45    """
 46    ml = 0
 47    mh = 0
 48    binl = 0
 49    binh = 0
 50    i = 0
 51    for edge in edges[:-1]:
 52        if edge < 0:
 53            binl = edge if hist[i] > ml else binl
 54            ml = max(ml, hist[i])
 55
 56        else:
 57            binh = edge if hist[i] > mh else binh
 58            mh = max(mh, hist[i])
 59
 60        i = i + 1
 61
 62    print(f'Max left {ml} @ {binl} \nMax right {mh} @ {binh}')
 63    print(f'Height: {binh - binl}')
 64
 65    return binh - binl

 68def _tolerant_mean(arrs: list):
 69    """
 70    Calculates the average between multiple arrays of different length
 71
 72    Parameters
 73    ----------
 74    arrs: list
 75        The arrays to be processed
 76
 77    Returns
 78    -------
 79    mean: np.array
 80        The mean calculated
 81    """
 82    lens = [len(i) for i in arrs]
 83    arr = np.ma.empty((np.max(lens), len(arrs)))
 84    arr.mask = True
 85    for idx, l in enumerate(arrs):
 86        arr[:len(l), idx] = l
 87    return arr.mean(axis=-1), arr.std(axis=-1)

120            for j in range(len(rois) - 2):  # consider j, j+1, j+2
121                outerMeanL = np.mean(rois[j].Z)
122                outerMeanR = np.mean(rois[j + 2].Z)

125                outerStdL = np.std(rois[j].Z)
126                outerStdR = np.std(rois[j + 2].Z)
127                innerStd = np.std(rois[j + 1].Z)

132                if outerStdL > abs(step) / 200 or outerStdR > abs(step) / 200 or innerStd > abs(step) / 200:

305    @staticmethod
306    def lateral(obj: profile.Profile, nom_pitch, bplt=False):
307        """
308        Evaluate RSM - N parameters with the sigmoid fit method
309        @dorothee_hueser
310
311        Parameters
312        ----------
313        obj: The profile on which the steps are evaluated
314        nom_pitch: the nominal pitch of the sample
315        bplt: if true 
316
317        Returns
318        -------
319        x_c: np.array
320            The center position of the features
321        h_c: np.array
322            The calculated heights of the features
323        """
324        x_c = []  # center positions of features
325        h_c = []  # height of features
326
327        # cosine fit for the whole profile
328        _cos = lambda p: 0.5 * p[0] * np.cos(np.pi * (obj.X - p[1]) / p[2]) + p[3] + p[4] * obj.X
329        _fs = lambda p, s, xsw, x: 1 / (1 + np.exp(s * (p[1] - x) - xsw))
330        _sigm = lambda p, x: p[0] * (_fs(p, 1, 1 / 2 * p[2], x) * _fs(p, -1, 1 / 2 * p[2], x) -
331                                     _fs(p, 1, 3 / 2 * p[2], x) * _fs(p, -1, 3 / 2 * p[2], x) + 1 / 2) + p[3]
332
333        zmax = np.max(obj.Z)
334        zmin = np.min(obj.Z)
335        p_init = np.array([zmax - zmin, 0, 0.5 * nom_pitch, 0.5 * (zmax + zmin), 0])
336        popt = optimize.leastsq(lambda p: _cos(p) - obj.Z, p_init)[0]
337        # print(f'Cosine period (sample pitch approx): {2 * popt[2]}')
338
339        if bplt:
340            fig, ax = plt.subplots()
341            ax.plot(obj.X, obj.Z, obj.X, _cos(popt))
342            ax.set_title(obj.name)
343
344        # first maximum is in p1, the following are in p1 + 2p2 * ip
345        period = 2 * popt[2]
346        for ip in range(1, int(max(obj.X) / period)):
347            xip = popt[1] + period * ip
348            boolbox = (xip - 3 / 2 * popt[2] < obj.X) & (obj.X < xip + 3 / 2 * popt[2])
349            xbox = obj.X[boolbox]  # x of the single box
350            zbox = obj.Z[boolbox]  # z of the single box
351
352            # now we can fit the sigmoid
353            p_init_sigm = np.array([np.ptp(obj.Z), xip,  period / 2, np.mean(obj.Z)])
354            popt_sigm = optimize.leastsq(lambda p: _sigm(p, xbox) - zbox, p_init_sigm)[0]
355
356            h = _sigm(popt_sigm, popt_sigm[1]) - _sigm(popt_sigm, popt_sigm[1] + popt_sigm[2])
357            d_form = popt_sigm[0] / h
358            th = 1
359            h_c.append(h if 1 - th < d_form < 1 + th else np.nan)
360            x_c.append(popt_sigm[1])
361
362            if bplt: ax.plot(xbox, _sigm(popt_sigm, xbox))
363
364        return np.array(x_c), np.abs(np.array(h_c))

524    def cylinder(obj: surface.Surface, radius, phiCone=None, alphaZ=0, concavity='convex', base=False, bplt=False):
525        """
526        Evaluates radius and form deviation of a cylinder by fitting a least square cylinder
527        to the points
528
529        Parametersq
530        ----------
531        obj: surface.Surface
532            The surface on which the processing is applied
533        radius: float
534            The nominal radius of the cylinder
535        phiCone: float
536            Angle in degree of the FOV of the instrument
537        alphaZ: rotation of the cylinder axis about the Y axis (radian)
538        concavity: str
539            Can be either 'convex' or 'concave'
540        base: bool
541            If true removes the points at the base of the cylinder
542        bplt: bool
543            Plots the sphere fitted to the data points
544
545        Returns
546        -------
547        R_all: float
548            The radius of the best fit cylinder to all points
549        FD_all: float
550            The form deviation of the best fit cylinder to all points
551        R_2s: float
552            The radius of the best fit cylinder to only the points with residue < 2 * sigma
553        FD_2s: float
554            The form deviation of the best fit cylinder to only the points with residue < 2 * sigma
555        """
556        def fitCyl():
557            _p = remover.Cylinder.remove(obj, radius, alphaZ=alphaZ, concavity=concavity,
558                                         base=base, finalize=False, bplt=bplt)
559            _r = np.abs(_p[0])
560
561            _l = np.cos(_p[3]) * np.cos(_p[4])
562            _m = np.sin(_p[3])
563            _n = np.cos(_p[3]) * np.sin(_p[4])
564
565            return _r, _p, _l, _m, _n
566
567        def calcResid():
568            # calculation of radial distance from fitted cylinder axis
569            d = -(l * obj.X + m * obj.Y + n * obj.Z)
570            t = -(d + m * est_p[1] + n * est_p[2])
571
572            H = np.array([l * t, est_p[1] + m * t, est_p[2] + n * t])
573            P = np.array([obj.X, obj.Y, obj.Z])
574
575            dist = np.linalg.norm(P - H, axis=0)
576
577            # calculation of radial residues as dist - fitR
578            _resid = dist - R
579
580            return _resid, np.mean(_resid[~np.isnan(_resid)]), np.std(_resid[~np.isnan(_resid)])
581
582        # fit the first approx cyl
583        R, est_p, l, m, n = fitCyl()
584        resid, avg, std = calcResid()
585
586        below_i = (np.abs(resid) < 2 * std)  # points with residues below 2sigma
587
588        if phiCone is not None:  # remove points outside cone from topo
589            base = R * np.sin(np.deg2rad(phiCone))
590            base = base if -np.pi/2 < est_p[3] < np.pi/2 else -base  # invert polarity for alphaZ >< +-90
591            # keep only values inside the range +- base centered on the cylinder axis
592            discard_i = np.abs(obj.Y - (m / l) * obj.X - est_p[1]) > (2 * base) / np.cos(est_p[3])
593            obj.Z[discard_i] = np.nan
594
595            if bplt: obj.pltC()
596
597        # fit the all cyl
598        R_all, est_p, l, m, n = fitCyl()
599        resid_all, avg_all, std_all = calcResid()
600        FD_all = np.ptp(resid_all[~np.isnan(resid_all)])
601
602        # fit the 2sigma cyl
603        obj.Z[~below_i] = np.nan
604        if bplt: obj.pltC()
605        R_2s, est_p, l, m, n = fitCyl()
606        resid_2s, avg_2s, std_2s = calcResid()
607        FD_2s = np.ptp(resid_2s[~np.isnan(resid_2s)])
608
609        return R_all, FD_all, R_2s, FD_2s

331        _sigm = lambda p, x: p[0] * (_fs(p, 1, 1 / 2 * p[2], x) * _fs(p, -1, 1 / 2 * p[2], x) -

336        p_init = np.array([zmax - zmin, 0, 0.5 * nom_pitch, 0.5 * (zmax + zmin), 0])
337        popt = optimize.leastsq(lambda p: _cos(p) - obj.Z, p_init)[0]
338        # print(f'Cosine period (sample pitch approx): {2 * popt[2]}')
339
340        if bplt:

342            ax.plot(obj.X, obj.Z, obj.X, _cos(popt))
343            ax.set_title(obj.name)
344
345        # first maximum is in p1, the following are in p1 + 2p2 * ip
346        period = 2 * popt[2]
347        for ip in range(1, int(max(obj.X) / period)):
348            xip = popt[1] + period * ip
349            boolbox = (xip - 3 / 2 * popt[2] < obj.X) & (obj.X < xip + 3 / 2 * popt[2])
350            xbox = obj.X[boolbox]  # x of the single box
351            zbox = obj.Z[boolbox]  # z of the single box
352
353            # now we can fit the sigmoid
354            p_init_sigm = np.array([np.ptp(obj.Z), xip,  period / 2, np.mean(obj.Z)])

362
363            if bplt: ax.plot(xbox, _sigm(popt_sigm, xbox))
364
365        return np.array(x_c), np.abs(np.array(h_c))

524    @staticmethod
525    def cylinder(obj: surface.Surface, radius, phiCone=None, alphaZ=0, concavity='convex', base=False, bplt=False):
526        """
527        Evaluates radius and form deviation of a cylinder by fitting a least square cylinder
528        to the points
529
530        Parametersq
531        ----------
532        obj: surface.Surface
533            The surface on which the processing is applied
534        radius: float
535            The nominal radius of the cylinder
536        phiCone: float
537            Angle in degree of the FOV of the instrument
538        alphaZ: rotation of the cylinder axis about the Y axis (radian)
539        concavity: str
540            Can be either 'convex' or 'concave'
541        base: bool
542            If true removes the points at the base of the cylinder
543        bplt: bool
544            Plots the sphere fitted to the data points
545
546        Returns
547        -------
548        R_all: float
549            The radius of the best fit cylinder to all points
550        FD_all: float
551            The form deviation of the best fit cylinder to all points
552        R_2s: float
553            The radius of the best fit cylinder to only the points with residue < 2 * sigma
554        FD_2s: float
555            The form deviation of the best fit cylinder to only the points with residue < 2 * sigma
556        """
557        def fitCyl():
558            _p = remover.Cylinder.remove(obj, radius, alphaZ=alphaZ, concavity=concavity,
559                                         base=base, finalize=False, bplt=bplt)
560            _r = np.abs(_p[0])
561
562            _l = np.cos(_p[3]) * np.cos(_p[4])
563            _m = np.sin(_p[3])
564            _n = np.cos(_p[3]) * np.sin(_p[4])
565
566            return _r, _p, _l, _m, _n
567
568        def calcResid():
569            # calculation of radial distance from fitted cylinder axis
570            d = -(l * obj.X + m * obj.Y + n * obj.Z)
571            t = -(d + m * est_p[1] + n * est_p[2])
572
573            H = np.array([l * t, est_p[1] + m * t, est_p[2] + n * t])
574            P = np.array([obj.X, obj.Y, obj.Z])
575
576            dist = np.linalg.norm(P - H, axis=0)
577
578            # calculation of radial residues as dist - fitR
579            _resid = dist - R
580
581            return _resid, np.mean(_resid[~np.isnan(_resid)]), np.std(_resid[~np.isnan(_resid)])
582
583        # fit the first approx cyl
584        R, est_p, l, m, n = fitCyl()
585        resid, avg, std = calcResid()
586
587        below_i = (np.abs(resid) < 2 * std)  # points with residues below 2sigma
588
589        if phiCone is not None:  # remove points outside cone from topo
590            base = R * np.sin(np.deg2rad(phiCone))
591            base = base if -np.pi/2 < est_p[3] < np.pi/2 else -base  # invert polarity for alphaZ >< +-90
592            # keep only values inside the range +- base centered on the cylinder axis
593            discard_i = np.abs(obj.Y - (m / l) * obj.X - est_p[1]) > (2 * base) / np.cos(est_p[3])
594            obj.Z[discard_i] = np.nan
595
596            if bplt: obj.pltC()
597
598        # fit the all cyl
599        R_all, est_p, l, m, n = fitCyl()
600        resid_all, avg_all, std_all = calcResid()
601        FD_all = np.ptp(resid_all[~np.isnan(resid_all)])
602
603        # fit the 2sigma cyl
604        obj.Z[~below_i] = np.nan
605        if bplt: obj.pltC()
606        R_2s, est_p, l, m, n = fitCyl()
607        resid_2s, avg_2s, std_2s = calcResid()
608        FD_2s = np.ptp(resid_2s[~np.isnan(resid_2s)])
609
610        return R_all, FD_all, R_2s, FD_2s
